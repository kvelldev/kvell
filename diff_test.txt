import { describe, it, expect } from "vitest";
import { detectSparkImage } from "@/domain/service/sparkUrlService";

describe("detectSparkImage", () => {
  it("returns null for text without URLs", () => {
    expect(detectSparkImage("Hello world")).toBeNull();
    expect(detectSparkImage("")).toBeNull();
  });

  it("returns null for non-whitelisted URLs", () => {
    // Generic domains are NOT allowed even with image extensions
    expect(detectSparkImage("https://example.com/page")).toBeNull();
    expect(detectSparkImage("https://unknown-site.com/image.png")).toBeNull();
    expect(
      detectSparkImage("https://s3.amazonaws.com/bucket/photo.png"),
    ).toBeNull();
  });

  it("detects Official domain images", () => {
    const url = "https://sakurazaka46.com/images/14/abc.jpg";
    expect(detectSparkImage(`Look: ${url}`)).toBe(url);
  });

  it("detects Twitter/X images (pbs.twimg.com)", () => {
    const url1 = "https://pbs.twimg.com/media/F_xxxxx.jpg";
    const url2 = "https://pbs.twimg.com/media/F_xxxxx?format=jpg&name=large";

    expect(detectSparkImage(url1)).toBe(url1);
    expect(detectSparkImage(`Wow ${url2} is great`)).toBe(url2);
  });

  it("detects Instagram CDN images", () => {
    const url =
      "https://scontent.cdninstagram.com/v/t51.2885-15/e35/p1080x1080/123.jpg";
    expect(detectSparkImage(url)).toBe(url);
  });

  it("generates YouTube thumbnail", () => {
    const ytUrl = "https://www.youtube.com/watch?v=dQw4w9WgXcQ";
    const expected = "https://img.youtube.com/vi/dQw4w9WgXcQ/hqdefault.jpg";
    expect(detectSparkImage(ytUrl)).toBe(expected);

    const shortening = "https://youtu.be/dQw4w9WgXcQ";
    expect(detectSparkImage(shortening)).toBe(expected);
  });

  it("ignores non-image URLs from whitelisted domains", () => {
    // Regex requires typical image structure for Official/Insta
    // Twitter regex requires /media/ path
    const invalidTw = "https://pbs.twimg.com/media/";
    expect(detectSparkImage(invalidTw)).toBeNull();
  });

  it("returns the first matching ALLOWED URL", () => {
    // example.com is not allowed, sakurazaka is allowed
    const safeUrl = "https://sakurazaka46.com/img.jpg";
    const unsafeUrl = "https://example.com/img.jpg";

    // Even if unsafe comes first, it should skipped (or return null if naive check)
    // Our logic iterates ALL urls and checks if any match whitelist.
    // It returns the FIRST one that matches.
    expect(detectSparkImage(`${unsafeUrl} and ${safeUrl}`)).toBe(safeUrl);
  });
});
